from ursina import *

class Player(Entity):
    def __init__(self):
        super().__init__(
            model='cube',
            color=color.orange,
            scale=(1, 2, 1),
            position=(0, 2, 0),
            collider='box'
        )
        self.velocity_y = 0
        self.gravity = -9.8
        self.jump_force = 10
        self.is_jumping = False

    def update(self):
        self.velocity_y += self.gravity * time.dt
        self.y += self.velocity_y * time.dt

        if self.y <= 1:  # Ground level
            self.y = 1
            self.velocity_y = 0
            self.is_jumping = False

        if held_keys['space'] and not self.is_jumping:
            self.velocity_y = self.jump_force
            self.is_jumping = True

        self.x += 5 * time.dt  # Constant forward motion

from ursina import *
import random

class Rooftop(Entity):
    def __init__(self, position):
        super().__init__(
            model='cube',
            color=color.gray,
            scale=(6, 1, 6),
            position=position,
            collider='box'
        )

class RooftopManager:
    def __init__(self):
        self.rooftops = []
        self.spawn_initial_rooftops()

    def spawn_initial_rooftops(self):
        for i in range(5):
            x = i * 8
            self.rooftops.append(Rooftop((x, 0, 0)))

    def update(self, player):
        last_rooftop = self.rooftops[-1]
        if player.x > last_rooftop.x - 30:
            new_x = last_rooftop.x + random.randint(6, 10)
            new_y = random.choice([0, 1, 2])
            self.rooftops.append(Rooftop((new_x, new_y, 0)))

        for rooftop in self.rooftops:
            if rooftop.x + 20 < player.x:
                destroy(rooftop)
                self.rooftops.remove(rooftop)

from ursina import *

class Player(Entity):
    def __init__(self):
        super().__init__(
            model='cube',
            color=color.orange,
            scale=(1, 2, 1),
            position=(0, 2, 0),
            collider='box'
        )
        self.velocity_y = 0
        self.gravity = -9.8
        self.jump_force = 10
        self.is_jumping = False

    def update(self):
        # Gravity and vertical movement
        self.velocity_y += self.gravity * time.dt
        self.y += self.velocity_y * time.dt

        # Cast a ray downward to check if standing on rooftop
        ray = raycast(self.world_position, direction=Vec3(0, -1, 0), distance=1.1, ignore=(self,), debug=False)

        if ray.hit:
            if self.velocity_y < 0:
                self.y = ray.world_point.y + self.scale_y / 2  # Place player on top of rooftop
                self.velocity_y = 0
                self.is_jumping = False
        else:
            self.is_jumping = True  # Player is in the air

        # Jumping
        if held_keys['space'] and not self.is_jumping:
            self.velocity_y = self.jump_force
            self.is_jumping = True

        # Constant forward movement
        self.x += 5 * time.dt
‚úÖ How It Works
raycast() checks straight down from the player's feet.

If the ray hits a rooftop within a small distance (1.1), it:

Stops falling

Resets jump

Snaps player to rooftop surface

If no roof below ‚Üí player keeps falling through the gap.

from ursina import *

class Player(Entity):
    def __init__(self):
        super().__init__(
            model='cube',
            color=color.orange,
            scale=(1, 2, 1),
            collider='box'
        )
        self.velocity_y = 0
        self.gravity = -0.1
        self.jump_force = 0.2
        self.on_ground = False
        self.double_jump_ready = False

    def update(self):
        ray = raycast(self.world_position, direction=Vec3(0, -1, 0), distance=1.1, ignore=(self,))
        self.on_ground = ray.hit

        if self.on_ground:
            self.velocity_y = 0
            self.double_jump_ready = True
            if held_keys['space']:
                self.velocity_y = self.jump_force
        else:
            if held_keys['space'] and self.double_jump_ready:
                self.velocity_y = self.jump_force
                self.double_jump_ready = False
            self.velocity_y += self.gravity

        self.y += self.velocity_y
        self.x += 0.1

from ursina import *
import random

class Rooftop(Entity):
    def __init__(self, position):
        super().__init__(
            model='cube',
            color=color.gray,
            scale=(6, 1, 6),
            position=position,
            collider='box'
        )
        self.spike = None
        if random.random() < 0.3:
            self.spike = Entity(
                model='cube',
                color=color.red,
                scale=(1, 1, 1),
                position=(position[0], position[1] + 1, position[2]),
                collider='box'
            )

class RooftopManager:
    def __init__(self, start_x=0):
        self.rooftops = []
        self.start_x = start_x
        self.spawn_initial_rooftops()

    def spawn_initial_rooftops(self):
        for i in range(5):
            x = self.start_x + i * 8
            self.rooftops.append(Rooftop((x, 0, 0)))

    def update(self, player):
        for rooftop in self.rooftops:
            if rooftop.x + 12 < player.x:
                rooftop.x += len(self.rooftops) * 8
                if rooftop.spike:
                    rooftop.spike.x = rooftop.x

        for i in range(5):
            x = -20 + i * 8

    def update(self, player):
